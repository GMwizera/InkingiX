SQL for top careers (school-scoped):
  SELECT c.name_en, c.name_kn, c.demand_level, COUNT(*) AS interest_count
  FROM career_matches cm
  JOIN careers c ON cm.career_id = c.id
  JOIN user_assessments ua ON cm.assessment_id = ua.id
  JOIN users u ON ua.user_id = u.id
  WHERE u.school_name = :school AND cm.rank = 1
  GROUP BY c.id
  ORDER BY interest_count DESC
  LIMIT 5;
